        // --- 2. SEO (DUMMY) DATA ---

        // Generates dummy keywords and scores based on the input topic

        function generateDummySEOData(topic) {

            const topicWords = topic.toLowerCase().split(' ').filter(w => w.length > 3);

           

            const keywords = [

                topic,

                ...topicWords.map(w => `best ${w}`),

                ...topicWords.map(w => `${w} tutorial`),

                ...topicWords.map(w => `how to ${w} 2025`),

                "youtube seo tips", "viral video strategy"

            ].slice(0, 5); // Limit to 5 keywords

            const tags = [

                topic.replace(/\s/g, ''),

                ...topicWords,

                "sunilcraft", "titleganai", "yt_seo", "youtube_hacks",

                ...keywords.map(k => k.replace(/\s/g, '_')).slice(0, 5)

            ].filter((v, i, a) => a.indexOf(v) === i).slice(0, 10); // Remove duplicates, limit to 10

            const keywordData = keywords.map(kw => ({

                keyword: toTitleCase(kw),

                score: Math.floor(Math.random() * 90) + 10 // Score 10-99

            }));

            return { keywordData, tags };

        }

       

        // --- 3. HELPER FUNCTIONS ---

        function getRandomItem(arr) {

            return arr[Math.floor(Math.random() * arr.length)];

        }

        function toTitleCase(str) {

            return str.replace(/\w\S*/g, txt =>

                txt.charAt(0).toUpperCase() + txt.substr(1)

            );

        }

        // Generates a single title based on the topic

        function generateTitle(topic) {

            const template = getRandomItem(templates);

            const year = Math.random() > 0.5 ? "2024" : "2025";

            const number = Math.floor(Math.random() * 20) + 1;

           

            let title = template

                .replace(/{topic}/g, toTitleCase(topic))

                .replace(/{power1}/g, getRandomItem(powerWords))

                .replace(/{power2}/g, getRandomItem(powerWords))

                .replace(/{phrase}/g, getRandomItem(phrases))

                .replace(/{year}/g, year)

                .replace(/{number}/g, number);

           

            return title;

        }

       

        // --- 4. CORE FUNCTION: RUN ANALYSIS ---

        function runAnalysis() {

            const input = document.getElementById("inputTitle").value.trim();

            if (!input) {

            // 1. Generate Titles

        // --- 5. DISPLAY FUNCTIONS ---

       

        function generateTitles(input) {

            const resultsContainer = document.getElementById("titles-results");

            resultsContainer.innerHTML = "";

           

            const titles = new Set();

            while (titles.size < 15) { // Generating 15 titles

                titles.add(generateTitle(input));

            }

            titles.forEach(title => {

                const div = document.createElement("div");

                div.className = "result-item";

               

                const titleSpan = document.createElement("span");

                titleSpan.textContent = title;

                div.appendChild(titleSpan);

               

                // Add event listener to update preview when a title is clicked/selected

                titleSpan.onclick = () => updatePreview(title);

                titleSpan.style.cursor = 'pointer';

                const copyButton = document.createElement("button");

                copyButton.textContent = "Copy";

               

                copyButton.onclick = function() {

                    navigator.clipboard.writeText(title).then(() => {

                        copyButton.textContent = "Copied!";

                        setTimeout(() => { copyButton.textContent = "Copy"; }, 1500);

                    }).catch(err => {

                         console.error('Could not copy text: ', err);

                         alert('Failed to copy. Please copy manually.');

                    });

                };

                div.appendChild(copyButton);

                resultsContainer.appendChild(div);

            });

        }

       

        function displayKeywords(keywordData) {

            const container = document.getElementById("keyword-results");

            container.innerHTML = "<h3>Keyword Score Legend: <span style='color: limegreen;'>Good (70+)</span>, <span style='color: yellow;'>Fair (40-69)</span>, <span style='color: orange;'>Low (10-39)</span></h3>";

           

            keywordData.forEach(item => {

                const itemDiv = document.createElement("div");

                itemDiv.className = "keyword-item";

               

                const scoreColor = item.score >= 70 ? 'limegreen' : item.score >= 40 ? 'yellow' : 'orange';

                const scoreFillColor = item.score >= 70 ? '#3CB371' : item.score >= 40 ? '#FFD700' : '#FF8C00';

                itemDiv.innerHTML = `

                    <div class="keyword-text">${item.keyword}</div>

                    <div class="score-bar">

                        <div class="score-fill" style="width: ${item.score}%; background-color: ${scoreFillColor};"></div>

                    </div>

                    <div class="keyword-score" style="color: ${scoreColor};">${item.score}/100</div>

                `;

                container.appendChild(itemDiv);

            });

        }

       

        function displayTags(tags) {

            const container = document.getElementById("tag-results");

            container.innerHTML = "";

           

            tags.forEach(tag => {

                const span = document.createElement("span");

                span.className = "tag-item";

                span.textContent = `#${tag}`;

                container.appendChild(span);

            });

           

            // Add copy all tags button

            const copyAllButton = document.createElement("button");

            copyAllButton.textContent = "Copy All Tags";

            copyAllButton.style.marginTop = '20px';

            copyAllButton.style.display = 'inline-block';

           

            copyAllButton.onclick = function() {

                const allTags = tags.map(tag => `#${tag}`).join(', ');

                navigator.clipboard.writeText(allTags).then(() => {

                    copyAllButton.textContent = "All Tags Copied!";

                    setTimeout(() => { copyAllButton.textContent = "Copy All Tags"; }, 1500);

                });

            };

            container.appendChild(copyAllButton);

        }
